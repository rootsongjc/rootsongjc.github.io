	if r.pluginID != id {

	}

func (r *pluginReference) ReferencesByName(ref reference.Named) []refstore.Association {
	return []refstore.Association{
		{
			Ref: r.name,
			ID:  r.pluginID,
		},
	}
}

	if r.name.String() != ref.String() {

	return r.pluginID, nil
